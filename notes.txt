void GenerateSalesReport(IEnumerable<string> salesFiles, string outputDirectory, string storesDirectory)
{
  double salesTotal = 0;

  string reportText = "Sales Summary" + Environment.NewLine;
  reportText += "----------------------------" + Environment.NewLine;

  List<string> details = new List<string>();

  foreach (var file in salesFiles)
  {
    string salesJson = File.ReadAllText(file);

    SalesData? data = JsonConvert.DeserializeObject<SalesData?>(salesJson);

    double fileTotal = data?.Total ?? data?.OverallTotal ?? 0;

    salesTotal += fileTotal;

    string fileName = Path.GetRelativePath(storesDirectory, file);

    details.Add($"{fileName}: {fileTotal:C}");
  }

  reportText += $"Total Sales: {salesTotal:C}" + Environment.NewLine + Environment.NewLine;

  reportText += "Details:" + Environment.NewLine;

  foreach (var detail in details)
  {
    reportText += detail + Environment.NewLine;
  }

  File.WriteAllText(
    Path.Combine(outputDirectory, "sales-summary.txt"),
    reportText
  );
}
